def f01(x, kT):
    return 1/(np.exp((x - 1)/kT) + 1)

kT = [5, 1, 0.2]
x = np.linspace(-20, 20, 100)
plt.figure(figsize=(10, 5))
for kTval in kT:
    f = [f01(xval, kTval) for xval in x]
    plt.plot(x, f, label='kT = '+str(kTval)+'$\mu$')
    plt.legend()
    plt.xlabel("$\\frac{\epsilon}{\mu}$", fontsize=14)
    plt.ylabel("$f_{FD}(\epsilon;\mu,T)$", fontsize=14)
plt.show()

f_MB = lambda x:1/np.exp(x/kT)

distribution = ['FD', 'BE', 'MB']
